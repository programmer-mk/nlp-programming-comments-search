/// Ova klasa predstavlja prioritetni red - FIFO princip.

public class PriorityQueue<T> where T : IComparable<T>
    {
        private readonly List<T> _pq = new List<T>();

        /// <summary>
        /// Dodavanje elementa u prioritetni red na njegov kraj.
        /// </summary>
        /// <param name="item"></param>
        public void Enqueue(T item)
        {
            _pq.Add(item);
        }

        /// <summary>
        /// Uklanjanje prvog elementa iz prioritetnog reda.
        /// </summary>
        /// <returns></returns>
        public T Dequeue()
        {
            var item = _pq[0];
            _pq.RemoveAt(0);

            return item;
        }
    }
