/// Konvertuj string u datum. Bezbedna konverzija.

public DateTime StringToDate(string value)
        {
            bool dateParsedSucessfully = DateTime.TryParse(value, out DateTime parsedDate);
            if (dateParsedSucessfully)
            {
                return parsedDate;
            }
            else 
            {
                return DateTime.Now;
            }
        }
