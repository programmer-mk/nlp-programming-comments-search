/// Klasa za konverziju json u xml

public class JsonToXMLConverter
    {
        /// <summary>
        /// Konvertuje json string u xml string.
        /// </summary>
        /// <param name="json"></param>
        /// <param name="rootName"></param>
        /// <returns></returns>
        public string ConvertJSONToXML(string json, string rootName = "Root")
        {
            try
            {
                XmlDocument node = JsonConvert.DeserializeXmlNode(json, rootName);
                return node.OuterXml.ToString();
            }
            catch (Exception e)
            {
                throw e;
            }
        }

        /// <summary>
        /// Konvertuje json u xml i cuva xml sadrzaj u fajlu.
        /// </summary>
        /// <param name="json"></param>
        /// <param name="fullFilePathToSave"></param>
        public void ConvertJson2Xml(string json, string fullFilePathToSave) 
        {
            try
            {
                XmlDocument xmlDocument = JsonConvert.DeserializeXmlNode(json);

                if (File.Exists(fullFilePathToSave) == false)
                {
                    File.Create(fullFilePathToSave);
                }

                XmlTextWriter xmlTextWriter = new XmlTextWriter(fullFilePathToSave, null);
                xmlTextWriter.Formatting = System.Xml.Formatting.Indented;
                xmlDocument.Save(xmlTextWriter);
            }
            catch (Exception e)
            {
                throw e;
            }
        }
