/// Ova metoda kreira niz zadate duzine i inicijalizuje ga random vrednostima izmedju minimalne i maksimalne vrednosti.

public int[] InitializeArrayRandom(int length, int minValue, int maxValue) 
        {
            var random = new Random();

            var array = new int[length];

            for (int i = 0; i < array.Length; i++)
            {
                array[i] = random.Next(minValue, maxValue + 1);
            }

            return array;
        }
