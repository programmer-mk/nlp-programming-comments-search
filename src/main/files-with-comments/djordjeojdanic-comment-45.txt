private void BrisanjeGeneriranihDokumenata()
{
            DirectoryInfo di = new DirectoryInfo(AppDomain.CurrentDomain.BaseDirectory + "Dokumenti");
            foreach (FileInfo file in di.GetFiles())
            {
                file.Delete();
            }
}
