/// Dobijanje datuma iz stringa koristeci format dd/MM/yyyy HH:mm:ss.ffffff.

public DateTime Str2Date(string strValue) 
        {
            try
            {
                DateTime output = DateTime.ParseExact(strValue, "dd/MM/yyyy HH:mm:ss.ffffff", CultureInfo.InvariantCulture);
                return output;            
            }
            catch (Exception exception)
            {
                throw exception;
            }
        }
