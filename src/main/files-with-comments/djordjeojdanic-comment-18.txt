/// <summary>
/// Slucajna vrednost po Poasonovoj raspodeli.
/// </summary>

public override double NextValue()
{
            double u = randObj.NextDouble();

            if (u == 1)
                u = 1.0 - double.Epsilon;

            double p = Math.Exp(-lambda);
            double f = p;
            int i = 0;

            while (f < u)
            {
                p = p * lambda / (i + 1);
                f = f + p;
                i++;
            }

            return i;
}
