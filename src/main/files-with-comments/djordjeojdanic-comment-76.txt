/// Nalazi sve brojeve u stringu i vraca ih kao niz.

public int[] NadjiSveBrojeveUString(string stringVrednost) 
        {
            string[] stringBrojevi = Regex.Split(stringVrednost, @"\D+").Where(str => string.IsNullOrWhiteSpace(str) == false).ToArray();

 

            int[] vrati = new int[stringBrojevi.Length];

 

            for (int i = 0; i < stringBrojevi.Length; i++)
            {
                vrati[i] = Convert.ToInt32(stringBrojevi[i]);
            }

 

            return vrati;
        }
