private void btnBubble_Click(object sender, EventArgs e)
{
    if (sorting)
        cancelThreads();

    sorting = true;
    sortThread = new Thread(() => elements.BubbleSort());
    sortThread.Start();

    if (!guest)
    {
        korisnik.bubble++;
        btnBubble.Text = "Bubble Sort: " + korisnik.bubble.ToString();
        _ctx.SaveChanges();
    }
}
