[HttpGet]
[Authorize]
[Route("get-user-data")]
public IHttpActionResult GetUserData()
{
    string userId = User.Identity.GetUserId();
    StatisticsUser user = _liftOffRepo.GetUserData(userId);

    if (user == null) return BadRequest();
    return Ok();
}
