// Vrati permutirani vektor

public Matrica VratiPermutiraniVektor(Matrica vektorPermutacije)
        {
            Matrica radniVektor = this.VratiKopiju();
            for (int i = 0; i < brojRedaka; i++)
            {
                radniVektor[i, 0] = this[(int)vektorPermutacije[0, i], 0];
            }
            return radniVektor;
        }
