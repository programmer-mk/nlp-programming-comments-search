/// <summary>
/// Metoda koja provjerava da li su sve niti u threadPool-u pokrenute.
/// </summary>

private bool isAllThreadsWork()
{
    bool flag = true;
    foreach (MyThreadForNonPreemptiveScheduler t in threadPool)
    {
        if (t.inRun == false)
            flag = false;
    }
    return flag;
}
