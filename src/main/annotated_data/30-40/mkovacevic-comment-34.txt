public void translateTo(int x, int y)
        {
            int changeY;

            if (y > this.Y)
            {
                changeY = 5;
            }
            else
            {
                changeY = -5;
            }

            while (y != this.Y)
            {
                this.Y += changeY;
                Thread.Sleep(150 - context.animationSpeed);
                context.Invalidate();
            }

            if (this.Y != y) this.Y = y;

            int changeX;

            if (x > this.X)
            {
                changeX = 5;
            }
            else
            {
                changeX = -5;
            }

            while (x != this.X)
            {
                this.X += changeX;
                Thread.Sleep(150 - context.animationSpeed);
                context.Invalidate();
            }

            if (this.X != x) this.X = x;
        }
