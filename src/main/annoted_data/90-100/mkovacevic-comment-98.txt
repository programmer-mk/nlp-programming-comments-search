// Promjeni izgled duha u njegovu standardnu formu kada lovi pacmana.
// Ne radi ništa ako duh bježi.

public void loadStandardImage()
{
    if (s == State.Flee) return;

    switch (currentDirection)
    {
        case Direction.None:
        case Direction.Up:
            currentImage = 0;
            break;
        case Direction.Left:
            currentImage = 1;
            break;
        case Direction.Down:
            currentImage = 2;
            break;
        case Direction.Right:
            currentImage = 3;
            break;
    }
}
