// rekurzivno dodavanje elementa u stablo

private void insertRec(Element e, Node n)
{
    if (e.Broj > n.inner.Broj)
    {
        if  (n.right == null)
        {
            n.right = new Node(e);
            return;
        }
        insertRec(e, n.right);
    }
    else
    {
        if (n.left == null)
        {
            n.left = new Node(e);
            return;
        }
        insertRec(e, n.left);
    }
}
