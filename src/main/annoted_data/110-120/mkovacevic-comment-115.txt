public void ateCherry(bool rotten)
{
    if (rotten)
    {
        isRottenCherryActive = true;
        rottenCherryDuration = Form1.RottenFruitDuration;
    }
    else
    {
        isCherryActive = true;
        cherryDuration = Form1.SuperCookieDuration;
    }
}
