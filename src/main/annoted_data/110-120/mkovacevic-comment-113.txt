// Dodavanje života.

public void addLife()
{
    // Život se dodaje samo ako ima manje od maxLives života.
    if (lives < maxLives)
    {
        // Dodajemo samo jedan život pa ažuriraj samo zadnju sličicu.
        livesPictureBoxes[lives].Image = characterImages[0];

        lives += 1;
    }


}
