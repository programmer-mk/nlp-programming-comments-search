// vraca cvor u blizini tacke (x, y); ako ne postoji, vraca null

public Cvor traziCvor(int x, int y)
{
    Cvor cvor = null;
    for (int i = 0; i < cvorovi.Count; i++)
    {
        if (cvorovi[i].pogodjen(x, y))
        {
            cvor = cvorovi[i];
            break;
        }
    }
    return cvor;
}
