// vraca pin kola sa identifikatorom Id; ako pin ne postoji, vraca null.

public Pin dajPin(int id)
{
    Pin pin = null;
    for (int I = 0; I < pinovi.Count; I++)
    {
        if (pinovi[I].dajID() == id)
        {
            pin = pinovi[I];
            break;
        }
    }
    return pin;
}
