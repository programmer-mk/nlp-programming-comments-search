protected override void OnPreviewKeyDown(PreviewKeyDownEventArgs e)
{
    e.IsInputKey = (e.KeyCode == Keys.Left || e.KeyCode == Keys.Right || e.KeyCode == Keys.Up || e.KeyCode == Keys.Down);
    base.OnPreviewKeyDown(e);
}
