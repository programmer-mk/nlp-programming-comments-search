public static int SuperCookieDuration
{
    get
    {
        int value = Math.Max(7000 - 350 * (Form1.pacman.Level - 1), 0);
        if (chosenGameMode == GameMode.Turbo) return value / 2;
        else return Math.Max(7000 - 350 * (Form1.pacman.Level - 1), 0);
    }
}
